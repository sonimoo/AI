```cs
string[] tests = {
    "ab",          // true, n=1, m=0
    "abab",        // true, n=2, m=0
    "abcd",        // true, n=1, m=1
    "ababcdcd",    // true, n=2, m=2
    "cd",          // false, n must be >=1
    "a",           // false
    "abb",         // false
    "abcdab",      // false, нельзя возвращаться к ab после начала cd
    "",            // false, n>=1
    "abcdcd",      // true, n=1, m=2
    "ababcd",      // true, n=2, m=1
    "ababc",       // false, не завершена пара в cd (после c нужен d)
};
```
